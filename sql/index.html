	`userpass` VARCHAR(50) NULL DEFAULT '0' COMMENT '密码',
	`name` VARCHAR(30) NULL DEFAULT '0' COMMENT '真实姓名',
	`roleId` INT(10) NULL DEFAULT '0' COMMENT '角色编号',
	`departmentId` INT(10) NULL DEFAULT '0' COMMENT '部门编号',

) COMMENT='用户表' COLLATE='utf8_general_ci' ENGINE=MyISAM AUTO_INCREMENT=132;
